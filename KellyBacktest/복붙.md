## 3. 전략 기반 켈리 백테스팅 (완전 흐름 - State → Event)

```bash
cd KellyBacktest
python -c "
from src import db_connector, signal_engine, strategy_analyzer, kelly_engine, backtest_engine

# 1. 데이터 로딩
prices = db_connector.get_prices_from_db(['005930'], start_date='2020-01-01', end_date='2024-12-31').iloc[:, 0]

# 2. 상태(State) 생성 → 이벤트(Event) 추출
state = signal_engine.golden_cross(prices, short=5, long=20)
events = strategy_analyzer.state_to_events(state, entry_state=1)

# 3. 신호 통계 분석 (Adjusted Kelly 기본)
trade_returns = strategy_analyzer.extract_signal_returns(prices, events, holding_period=20, direction='long')
stats = strategy_analyzer.compute_kelly_params(trade_returns)

print('Trades:', stats['n_trades'])
print('Win rate:', f\"{stats['win_rate']*100:.1f}%\")
print('Avg win:', f\"{stats['avg_win']*100:.2f}%\")
print('Avg loss:', f\"{stats['avg_loss']*100:.2f}%\")
print('Adjusted f*:', f\"{stats['f_star_adjusted']*100:.1f}%\")

# 4. 켈리 사이즈 계산 (Half Kelly)
f = max(0.0, kelly_engine.fractional_kelly(stats['f_star_adjusted'], 0.5))
print('Half Kelly:', f\"{f*100:.1f}%\")

# 5. 백테스트 실행 (Next-Day Execution)
result = backtest_engine.run_strategy(prices, events, kelly_fraction=f, holding_period=20)
print('Kelly NAV:', round(result['nav'].iloc[-1], 0))
print('Buy&Hold NAV:', round(result['buyhold'].iloc[-1], 0))
print('Trades executed:', len(result['trade_log']))
"
```

## 4. 다양한 전략 테스트 (Event-based)

```bash
cd KellyBacktest
python -c "
from src import db_connector, signal_engine, strategy_analyzer

prices = db_connector.get_prices_from_db(['000660'], start_date='2020-01-01').iloc[:, 0]

for name, fn in [
    ('Golden Cross', lambda p: signal_engine.golden_cross(p, 5, 20)),
    ('RSI', lambda p: signal_engine.rsi_signal(p, 14, 30, 70)),
    ('Momentum', lambda p: signal_engine.momentum_breakout(p, 20)),
    ('Bollinger', lambda p: signal_engine.bollinger_signal(p, 20, 2)),
]:
    state = fn(prices)
    events = strategy_analyzer.state_to_events(state, entry_state=1)
    returns = strategy_analyzer.extract_signal_returns(prices, events, holding_period=20)
    stats = strategy_analyzer.compute_kelly_params(returns)
    status = 'POSITIVE' if stats['f_star_adjusted'] > 0 else 'NEGATIVE'
    print(f'{name}: {status} | f_adj={stats[\"f_star_adjusted\"]*100:.1f}% | Win={stats[\"win_rate\"]*100:.1f}% | Trades={stats[\"n_trades\"]}')
"
```

## 5. Grid Search (자동 파라미터 최적화)

```bash
cd KellyBacktest
python -c "
from src import db_connector, strategy_analyzer

prices = db_connector.get_prices_from_db(['005930'], start_date='2020-01-01').iloc[:, 0]

grid = strategy_analyzer.grid_search(
    prices,
    signal_name='golden_cross',
    param_grid={'short': [3, 5, 10], 'long': [20, 30, 60]},
    holding_periods=[10, 20, 40],
    min_trades=30,
    min_f_star=0.0,
)

print('Top 5 combinations:')
for i, row in grid.head(5).iterrows():
    print(f\"#{i+1} f_adj={row['f_star_adjusted']*100:.1f}% | Win={row['win_rate']*100:.1f}% | short={row['short']} long={row['long']} HP={row['holding_period']}\")
"
```

## 6. OOS (Out-of-Sample) 분리 테스트

```bash
cd KellyBacktest
python -c "
from src import db_connector, signal_engine, strategy_analyzer, backtest_engine

prices = db_connector.get_prices_from_db(['005930'], start_date='2020-01-01').iloc[:, 0]
state = signal_engine.golden_cross(prices, 5, 20)
events = strategy_analyzer.state_to_events(state, entry_state=1)

is_stats, f_star, oos_returns = strategy_analyzer.analyze_with_oos(
    prices, events, split_ratio=0.7, holding_period=20
)

print('IS f*:', f\"{f_star*100:.1f}%\")
oos_stats = strategy_analyzer.compute_kelly_params(oos_returns)
print('OOS f*:', f\"{oos_stats['f_star_adjusted']*100:.1f}%\")
print('OOS trades:', oos_stats['n_trades'])

# OOS 백테스트
result = backtest_engine.run_strategy(
    prices.iloc[int(len(prices)*0.7):],
    events.iloc[int(len(prices)*0.7):],
    kelly_fraction=f_star,
    holding_period=20,
)
print('OOS Kelly NAV:', round(result['nav'].iloc[-1], 0))
"
```

## 7. 샘플 데이터로 테스트

```bash
cd KellyBacktest
python -c "
from src import data_loader, signal_engine, strategy_analyzer, kelly_engine, backtest_engine

prices = data_loader.generate_sample_data(years=3, mu=0.1, sigma=0.2, seed=42)
prices = data_loader.validate_data(prices)

state = signal_engine.golden_cross(prices, 5, 20)
events = strategy_analyzer.state_to_events(state, entry_state=1)
trade_returns = strategy_analyzer.extract_signal_returns(prices, events, holding_period=20)

result = backtest_engine.run_strategy(prices, events, kelly_fraction=f, holding_period=20)

print('Adjusted f*:', f\"{stats['f_star_adjusted']*100:.1f}%\")
print('NAV:', round(result['nav'].iloc[-1], 0))
print('Trades:', len(result['trade_log']))
"
```
